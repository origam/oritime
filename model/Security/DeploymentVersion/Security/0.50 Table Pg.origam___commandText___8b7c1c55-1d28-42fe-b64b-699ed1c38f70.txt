	"LastLoginDate" Timestamp(3) NULL,
	"refBusinessPartnerId" Uuid NOT NULL,
	"IsLockedOut" Boolean NOT NULL,
	"LastLockoutDate" Timestamp(3) NULL,
	"UserName" Varchar(255) NULL,
	"Is2FAEnforced" Boolean NOT NULL,
